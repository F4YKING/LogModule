	LogModule API

	.AddLog(LogMessage: string?)
	Adds log message to the data.
	
	.PrintLogs()
	Prints log messages to console/output.
	
	.SendLogs(plr: player)
	Send log(s) to discord with webhook(s).
	
	.CheckAdmin(plr: Player)
	Check if player is admin or not, if admin; player get access to Send Logs by saying the SendLogsCommand.
}
